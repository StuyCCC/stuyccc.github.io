Solution C++ 

#include <bits/stdc++.h>
using namespace std;

int main() {
    int t; cin >> t;
    for (int z = 0; z < n; z++) {
        int n; cin >> n;
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
            cin >> arr[i];
        }
        int m = arr[0];
        for (int i = 0; i < n; i++) {
            m = min(m, arr[i]);
        }
        int ans = 0;
        for (int i = 0; i < n; i++) {
            arr[i] -= m;
            ans += arr[i];
        }
        cout << ans << '\n';
    }
}

Increasing Array, CSES

Solution C++ 

#include <bits/stdc++.h>
using namespace std;

int main() {
        int n; cin >> n;
        vector<int> arr(n);
        for (int i = 0; i < n; i++) {
                cin >> arr[i];
        }

        long long ans = 0;
        for (int i = 1; i < n; i++) {
                if (arr[i-1] > arr[i]) {
                        ans += arr[i-1] - arr[i];
                        arr[i] = arr[i-1];
                }
        }
        cout << ans << '\n';
}
Solution Python

ans = 0

        ls[i] = ls[i-1]
